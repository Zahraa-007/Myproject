Digital root is the recursive sum of all the digits in a number.


int digital_root(int n) {
    int s = 0;
    while (n != 0) {
        s += n % 10;
        n /= 10;
    }
    if (s >= 10)
        s = digital_root(s);
    return s;
}

Given an array of integers, construct a product array such that each element is the product of all other elements.


#include <vector>

std::vector<long long> productArray(const std::vector<int>& numbers) {
    int n = numbers.size();
    std::vector<long long> arr(n, 1);
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            if (j != i)
                arr[i] *= numbers[j];
        }
    }
    return arr;
}

Check if a string is a palindrome


#include <string>

bool isPalindrome(const std::string& str) {
    int left = 0;
    int right = str.length() - 1;
    while (left < right) {
        if (str[left] != str[right])
            return false;
        left++;
        right--;
    }
    return true;
}

Find the largest element in an array


int findMax(const std::vector<int>& arr) {
    int max = arr[0];
    for (int i = 1; i < arr.size(); i++) {
        if (arr[i] > max)
            max = arr[i];
    }
    return max;
}
