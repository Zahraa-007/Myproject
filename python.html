Digital root is the recursive sum of all the digits in a number.


def digital_root(n):
    s = 0
    while n != 0:
        s += n % 10
        n //= 10
    if s >= 10:
        s = digital_root(s)
    return s

Given an array of integers, construct a product array such that each element is the product of all other elements.


def product_array(numbers):
    arr = []
    for i in range(len(numbers)):
        prod = 1
        for j in range(len(numbers)):
            if i != j:
                prod *= numbers[j]
        arr.append(prod)
    return arr

Find the largest element in an array


def find_max(arr):
    max_val = arr[0]
    for i in range(1, len(arr)):
        if arr[i] > max_val:
            max_val = arr[i]
    return max_val
